from pyspark.sql.functions import col, when

#Writing function for replacing blacks with Null
def blank_as_null(x):
    return when(col(x) != "", col(x)).otherwise(None)

to_convert = set(["Rating","Current Ver","Android Ver"]) # Some set of columns
